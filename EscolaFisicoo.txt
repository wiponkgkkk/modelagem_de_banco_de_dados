create table estuda ( id_estuda int primary key auto_increment, fk_Aluno_matricula int not null, fk_Turma_numero int not null );

alter table estuda add constraint fkmat foreign key (fk_Aluno_matricula) references Aluno(matricula);

alter table estuda add constraint fkturma foreign key (fk_Turma_numero) references Turma (numero);

create table Possui (id_possui int primary key, fk_Disciplina_cod int not null, fk_Turma_numero int not null);

alter table Possui add constraint fknum foreign key (fk_Turma_numero) references Disciplina (cod_disciplina);

alter table Possui add constraint fkdisc foreign key (fk_Disciplina_cod) references Turma (numero);

create table Professor (matricula int primary key auto_increment, formacao varchar(30), nome varchar(80) not null);

create table Leciona (id_leciona int primary key auto_increment, fk_Disciplina_cod_disciplina int not null, fk_Professor_matricula int not null);

alter table Leciona add constraint fkdis foreign key (fk_Disciplina_cod_disciplina) references Disciplina (cod_disciplina);

alter table Leciona add constraint fkprof foreign key (fk_Professor_matricula) references Professor (matricula);
